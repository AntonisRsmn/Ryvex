function getLogChannel(guild, settings, type = "general") {
  if (type === "moderation") {
    const modChannelId = settings.moderation?.channelId;
    if (modChannelId) {
      return guild.channels.cache.get(modChannelId) ?? null;
    }
  }

  // Fallback to general logging channel
  const generalChannelId = settings.logging?.channelId;
  if (!generalChannelId) return null;

  return guild.channels.cache.get(generalChannelId) ?? null;
}

module.exports = { getLogChannel };
